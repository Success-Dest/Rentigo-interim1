-- Drop existing table
DROP TABLE IF EXISTS market_properties;

-- Create market_properties table
CREATE TABLE market_properties (
    id INT PRIMARY KEY AUTO_INCREMENT,
    landlord_id INT NOT NULL DEFAULT 1,
    manager_id INT NULL,
    address TEXT NOT NULL,
    property_type ENUM('apartment', 'house', 'condo', 'townhouse') NOT NULL,
    bedrooms INT NOT NULL,
    bathrooms DECIMAL(2,1) NOT NULL,
    sqft INT NULL,
    rent DECIMAL(10,2) NOT NULL,
    deposit DECIMAL(10,2) NULL,
    description TEXT NULL,
    status ENUM('available', 'occupied', 'maintenance', 'vacant', 'pending') DEFAULT 'available',
    available_date DATE NULL,
    parking VARCHAR(50) DEFAULT '0',
    pet_policy ENUM('no', 'cats', 'dogs', 'both') NOT NULL DEFAULT 'no',
    laundry ENUM('none', 'shared', 'in_unit', 'hookups', 'included') NOT NULL DEFAULT 'none',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_type_bedrooms (property_type, bedrooms),
    INDEX idx_rent (rent),
    INDEX idx_status (status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

SELECT '=== TOTAL PROPERTIES ===' as Info;
SELECT COUNT(*) as Total_Properties FROM market_properties;

-- Property type distribution
SELECT '=== PROPERTY TYPE DISTRIBUTION ===' as Info;
SELECT property_type, COUNT(*) as count FROM market_properties GROUP BY property_type;

-- Bedroom distribution
SELECT '=== BEDROOM DISTRIBUTION ===' as Info;
SELECT bedrooms, COUNT(*) as count FROM market_properties GROUP BY bedrooms ORDER BY bedrooms;

-- Rent statistics
SELECT '=== RENT STATISTICS ===' as Info;
SELECT 
    MIN(rent) as Min_Rent,
    ROUND(AVG(rent), 2) as Avg_Rent,
    MAX(rent) as Max_Rent
FROM market_properties;

-- Status distribution
SELECT '=== STATUS DISTRIBUTION ===' as Info;
SELECT status, COUNT(*) as count FROM market_properties GROUP BY status;

-- Area distribution
SELECT '=== COLOMBO AREA DISTRIBUTION ===' as Info;

        WHEN address LIKE '%Colombo 11%' THEN 'Colombo 11'
        WHEN address LIKE '%Colombo 12%' THEN 'Colombo 12'
        WHEN address LIKE '%Colombo 13%' THEN 'Colombo 13'
        WHEN address LIKE '%Colombo 14%' THEN 'Colombo 14'
        WHEN address LIKE '%Colombo 15%' THEN 'Colombo 15'

FROM market_properties
GROUP BY Colombo_Area
ORDER BY Colombo_Area;
